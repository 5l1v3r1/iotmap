# This function takes the handle and returns a dictionnary with
# the handle value and the characteristics associated.
# Especially those properties :
# - Characteristic properties
# - Characteristic Value Handle
# - UUID
# The value is in byte format
def getHandles(handle, verbose):
    hexValue = handle.value.hex()
    charProperty = hexValue[:2]
    valueHandle = hexValue[4:6] + hexValue[2:4]
    UUID = hexValue[6:] if len(hexValue) > 10 else hexValue[8:10] + hexValue[6:8]
    
    if len(UUID) == 4:
        if verbose:
            if UUID in UUIDS_16BITS.keys():
                UUID = UUIDS_16BITS[int(UUID, 16)]

    else :
        UUID = getUUIDFormat(UUID)
        if verbose:
            if UUID in UUIDS_128BITS.keys():
                UUID = UUIDS_128BITS[UUID]


    characteristics = {
        'properties': getProperties(int(charProperty, 16)) if verbose else charProperty,
        'Value Handle': valueHandle,
        'UUID': UUID
    }

    ret = {
        'handle': handle.handle,
        'value': characteristics
    }
    
    return ret

def getUUIDFormat(uuid):
    UUID = uuid[-2:]
    for i in range(2, len(uuid), 2):
        UUID += uuid[-i-2:-i]

    UUID = UUID[:8] + '-' + UUID[8:12] + '-' + UUID[12:16] + '-' + UUID[16:20] + '-' + UUID[20:]
    
    return UUID

def getFlags(flag):
    ret = []
    mask = 1
    for k in flags.keys():
        if flags[k] == 'RFU1': 
            mask = 7 # The mask is 111
        if flags[k] == 'RFU2': 
            mask = 3 # The mask 11

        # Check if the current flag is on
        if (flag >> k) & mask:
            ret.append(flags[k])

    return ret
